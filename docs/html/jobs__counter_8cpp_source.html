106 result counter_handle::wait_for(size_t value, timeout in_timeout)
107 {
108  jobs_profile_scope(profile_scope_type::fiber, "counter::wait_for", m_scheduler);
109 
110  internal::counter_definition& def = m_scheduler->get_counter_definition(m_index);
111 
112  // Grab the current job context.
113  internal::job_context* context = m_scheduler->get_active_job_context();
114  internal::job_context* worker_context = m_scheduler->get_worker_job_context();
115 
116  // If we have a job context, go to sleep waiting on the value.
117  if (context != nullptr)
118  {
119  assert(worker_context != nullptr);
120 
121  volatile bool timeout_called = false;
122 
123  // Put job to sleep.
124  {
125  context->job_def->status = internal::job_status::waiting_on_counter;
126  context->job_def->wait_counter = *this;
127  context->job_def->wait_counter_value = value;
128  context->job_def->wait_counter_remove_value = false;
129  context->job_def->wait_counter_do_not_requeue = false;
130 
131  if (!add_to_wait_list(context->job_def))
132  {
133  return result::success;
134  }
135  }
136 
137  // Queue a wakeup.
138  size_t schedule_handle;
139  if (!in_timeout.is_infinite())
140  {
141  result res = m_scheduler->m_callback_scheduler.schedule(in_timeout, schedule_handle, [&]() {
142 
143  // Do this atomatically to make sure we don't set it to pending after the scheduler
144  // has already done that due to this event being signalled.
145  internal::job_status expected = internal::job_status::waiting_on_counter;
146  if (context->job_def->status.compare_exchange_strong(expected, internal::job_status::pending))
147  {
148  timeout_called = true;
149  m_scheduler->requeue_job(context->job_def->index);
150  }
151 
152  });
153 
154  // Failed to schedule a wakeup? Abort.
155  if (res != result::success)
156  {
157  remove_from_wait_list(context->job_def);
158  context->job_def->status = internal::job_status::pending;
159  return res;
160  }
161  }
162 
163  // Supress requeueing the job, and return to worker, we will do this when the callback returns.
164  m_scheduler->return_to_worker(*worker_context, true);
165 
166  // Cleanup
167  context->job_def->wait_counter = counter_handle();
168 
169  // If we timed out, just escape here.
170  if (timeout_called)
171  {
172  return result::timeout;
173  }
174 
175  // Cancel callback and clear our job handle.
176  if (!in_timeout.is_infinite())
177  {
178  m_scheduler->m_callback_scheduler.cancel(schedule_handle);
179  }
180 
181  return result::success;
182  }
183 
184  // If no job-context we have to do a blocking wait.
185  else
186  {
187  internal::stopwatch timer;
188  timer.start();
189 
190  {
191  std::unique_lock<std::mutex> lock(def.value_cvar_mutex);
192 
193  // Allocate a fake job to be waited on.
194  internal::job_definition fake_job(UINT32_MAX);
195  fake_job.status = internal::job_status::waiting_on_counter;
196  fake_job.wait_counter = *this;
197  fake_job.wait_counter_value = value;
198  fake_job.wait_counter_remove_value = false;
199  fake_job.wait_counter_do_not_requeue = true;
200 
201  if (!add_to_wait_list(&fake_job))
202  {
203  return result::success;
204  }
205 
206  // Keep looping until we manage to decrease the value.
207  while (fake_job.status == internal::job_status::waiting_on_counter)
208  {
209  if (in_timeout.is_infinite())
210  {
211  def.value_cvar.wait(lock);
212  }
213  else
214  {
215  size_t ms_remaining = in_timeout.duration - timer.get_elapsed_ms();
216  if (ms_remaining > 0)
217  {
218  def.value_cvar.wait_for(lock, std::chrono::milliseconds(ms_remaining));
219  }
220  else
221  {
222  return result::timeout;
223  }
224  }
225  }
226  }
227  }
228 
229  return result::success;
230 }

383 bool counter_handle::modify_value(size_t new_value, bool absolute, bool subtract, bool lock_required)
384 {
385  jobs_profile_scope(profile_scope_type::fiber, "counter::modify_value", m_scheduler);
386 
387  internal::counter_definition& def = m_scheduler->get_counter_definition(m_index);
388 
389  {
390  internal::optional_shared_lock<internal::spinwait_mutex> lock(def.wait_list.get_mutex(), lock_required);
391 
392  size_t changed_value = 0;
393  if (absolute)
394  {
395  changed_value = (def.value = new_value);
396  }
397  else
398  {
399  if (subtract)
400  {
401  if (def.value >= new_value)
402  {
403  changed_value = (def.value -= new_value);
404  }
405  else
406  {
407  return false;
408  }
409  }
410  else
411  {
412  changed_value = (def.value += new_value);
413  }
414  }

416  notify_value_changed(changed_value, false);
417  }

422 bool counter_handle::add_to_wait_list(internal::job_definition* job_def)
423 {
424  jobs_profile_scope(profile_scope_type::fiber, "counter::add_to_wait_list", m_scheduler);
425 
426  internal::counter_definition& def = m_scheduler->get_counter_definition(m_index);
427 
428  if (job_def->wait_counter_remove_value)
429  {
430  internal::optional_lock<internal::spinwait_mutex> lock(def.wait_list.get_mutex());
431 
432  // Value high enough to remove? We don't need to wait.
433  if (modify_value(job_def->wait_counter_value, false, true, false))
434  {
435  job_def->status = internal::job_status::pending;
436  job_def->wait_counter = counter_handle();
437 
438  return false;
439  }
440 
441  job_def->wait_counter_list_link.value = job_def;
442  def.wait_list.add(&job_def->wait_counter_list_link, false);
443  }
444  else
445  {
446  internal::optional_shared_lock<internal::spinwait_mutex> lock(def.wait_list.get_mutex());
447 
448  // Is value equal? We don't need to wait.
449  if (def.value == job_def->wait_counter_value)
450  {
451  job_def->status = internal::job_status::pending;
452  job_def->wait_counter = counter_handle();
453 
454  return false;
455  }
456 
457  job_def->wait_counter_list_link.value = job_def;
458  def.wait_list.add(&job_def->wait_counter_list_link, false);
459  }
460 
461  return true;

473 void counter_handle::notify_value_changed(size_t new_value, bool lock_required)
474 {
475  jobs_profile_scope(profile_scope_type::fiber, "counter::notify_value_changed", m_scheduler);
476 
477  internal::counter_definition& def = m_scheduler->get_counter_definition(m_index);
478 
479  // Go through wait list and mark all jobs as ready to resume if their criteria is met.
480  size_t signalled_job_count = 0;
481  size_t signalled_no_requeue_job_count = 0;
482 
483  {
484  jobs_profile_scope(profile_scope_type::fiber, "wake up waiting", m_scheduler);
485 
486  internal::multiple_writer_single_reader_list<internal::job_definition*>::iterator iter;
487  for (def.wait_list.iterate(iter, lock_required); iter; )
488  {
489  bool signal = false;
490 
491  internal::job_definition* job_def = iter.value();
492 
493  if (job_def->wait_counter_remove_value)
494  {
495  signal = modify_value(job_def->wait_counter_value, false, true, false);
496  }
497  else
498  {
499  signal = (new_value == job_def->wait_counter_value);
500  }
501 
502  if (signal)
503  {
504  internal::job_status expected = internal::job_status::waiting_on_counter;
505  if (job_def->status.compare_exchange_strong(expected, internal::job_status::pending))
506  {
507  if (!job_def->wait_counter_do_not_requeue)
508  {
509  m_scheduler->requeue_job(job_def->index);
510  signalled_job_count++;
511  }
512  else
513  {
514  signalled_no_requeue_job_count++;
515  }
516 
517  iter.remove();
518  continue;
519  }
520  }
521 
522  iter++;
523  }
524  }
525 
526  if (signalled_job_count > 0)
527  {
528  jobs_profile_scope(profile_scope_type::fiber, "signal worker threads", m_scheduler);
529  m_scheduler->notify_job_available(signalled_job_count);
530  }
531 
532  if (signalled_no_requeue_job_count > 0)
533  {
534  jobs_profile_scope(profile_scope_type::fiber, "signal waiting threads", m_scheduler);
535 
536  std::unique_lock<std::mutex> lock(def.value_cvar_mutex);
537  def.value_cvar.notify_all();
538  }
539 }
