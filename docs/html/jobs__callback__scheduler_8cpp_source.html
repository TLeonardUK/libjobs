52 result callback_scheduler::init(jobs::scheduler* scheduler, size_t max_callbacks, const jobs::memory_functions& memory_functions)
53 {
54  m_scheduler = scheduler;
55  m_memory_functions = memory_functions;
56 
57  // Init callback pool.
58  result result = m_callback_pool.init(m_memory_functions, max_callbacks, [this](callback_definition* instance, size_t index)
59  {
60  new(instance) callback_definition();
61  return result::success;
62  });
63  if (result != result::success)
64  {
65  return result;
66  }
67 
68  // Init main thread.
69  m_callback_thread = (thread*)m_memory_functions.user_alloc(sizeof(thread), alignof(thread));
70  new(m_callback_thread) thread(m_memory_functions);
71 
72  m_callback_thread->init([&]()
73  {
74  while (!m_shutting_down)
75  {
76  std::unique_lock<std::mutex> lock(m_schedule_updated_mutex);
77 
78  run_callbacks();
79 
80  // No callbacks? Wait until we have one.
81  uint64_t time_till_next_callback = get_ms_till_next_callback();
82  if (m_callback_pool.count() == 0 || time_till_next_callback == UINT64_MAX)
83  {
84  m_schedule_updated_cvar.wait(lock);
85  }
86  // Wait until next callback is due to run.
87  else
88  {
89  m_schedule_updated_cvar.wait_for(lock, std::chrono::milliseconds(time_till_next_callback));
90  }
91  }
92  }, "Latent Callback Scheduler", 0xFFFF);
93 
94  return result::success;

97 void callback_scheduler::run_callbacks()
98 {
99  // @todo: this has unneccessary iteration over free elements, need to maintain an allocated list.
100  for (size_t i = 0; i < m_callback_pool.capacity(); i++)
101  {
102  callback_definition& def = *m_callback_pool.get_index(i);
103  if (def.active)
104  {
105  if (def.stopwatch.get_elapsed_ms() >= def.duration.duration)
106  {
107  def.callback();
108 
109  size_t handle = i + (def.generation * m_callback_pool.capacity());
110 
111  def.callback = nullptr;
112  def.active = false;
113 
114  m_callback_pool.free(i);
115  }
116  }
117  }
118 }

120 uint64_t callback_scheduler::get_ms_till_next_callback()
121 {
122  uint64_t time_till_next = UINT64_MAX;
123 
124  // @todo: this has unneccessary iteration over free elements, need to maintain an allocated list.
125  for (size_t i = 0; i < m_callback_pool.capacity(); i++)
126  {
127  callback_definition& def =* m_callback_pool.get_index(i);
128  if (def.active && def.duration.duration != timeout::infinite.duration)
129  {
130  size_t remaining = JOBS_MAX(0, def.duration.duration - def.stopwatch.get_elapsed_ms());
131  time_till_next = JOBS_MIN(time_till_next, remaining);
132  }
133  }
134 
135  return time_till_next;

138 result callback_scheduler::schedule(timeout duration, size_t& handle, const callback_scheduler_function& callback)
139 {
140  std::unique_lock<std::mutex> lock(m_schedule_updated_mutex);
141 
142  size_t index = 0;
143  result res = m_callback_pool.alloc(index);
144  if (res != result::success)
145  {
146  m_scheduler->write_log(debug_log_verbosity::warning, debug_log_group::scheduler, "attempt to create latent callback, but pool is empty. Behaviour may be incorrect. Try increasing scheduler::set_max_callbacks.");
147  return res;
148  }
149 
150  callback_definition* def = m_callback_pool.get_index(index);
151  def->active = true;
152  def->stopwatch.start();
153  def->duration = duration;
154  def->callback = callback;
155  def->generation++;
156 
157  m_schedule_updated_cvar.notify_all();
158 
159  handle = index + (def->generation * m_callback_pool.capacity());
160 
161  return result::success;
162 }

164 result callback_scheduler::cancel(size_t handle)
165 {
166  std::unique_lock<std::mutex> lock(m_schedule_updated_mutex);
167 
168  size_t index = handle % m_callback_pool.capacity();
169  size_t generation = (handle / m_callback_pool.capacity());
170 
171  callback_definition* def = m_callback_pool.get_index(index);
172  if (def->generation != generation)
173  {
174  return result::already_complete;
175  }
176 
177  def->callback = nullptr;
178  def->active = false;
179 
180  m_callback_pool.free(index);
181 
182  return result::success;
183 }
