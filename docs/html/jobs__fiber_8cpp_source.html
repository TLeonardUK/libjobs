82 result fiber::init(size_t stack_size, const fiber_entry_point& entry_point, const char* name)
83 {
84  m_entry_point = entry_point;
85 
86 #if defined(JOBS_PLATFORM_WINDOWS) || defined(JOBS_PLATFORM_XBOX_ONE)
87 
88  m_fiber_handle = CreateFiberEx(stack_size, stack_size, FIBER_FLAG_FLOAT_SWITCH, trampoline_entry_point, this);
89  if (m_fiber_handle == nullptr)
90  {
91  // Try and figure out some useful results if possible.
92  DWORD error = GetLastError();
93  if (error == ERROR_OUTOFMEMORY)
94  {
95  return result::out_of_memory;
96  }
97 
98  return result::platform_error;
99  }
100 
101 #elif defined(JOBS_PLATFORM_PS4)
102 
103  m_fiber_context = m_memory_functions.user_alloc(stack_size, SCE_FIBER_CONTEXT_ALIGNMENT);
104  if (m_fiber_context == nullptr)
105  {
106  return result::out_of_memory;
107  }
108 
109  int32_t ret = sceFiberInitialize(&m_fiber_handle, name, trampoline_entry_point, reinterpret_cast<uint64_t>(this), m_fiber_context, stack_size, nullptr);
110  if (ret != SCE_OK)
111  {
112  return result::platform_error;
113  }
114 
115  m_fiber_handle_created = true;
116 
117 #elif defined(JOBS_PLATFORM_SWITCH)
118 
119  // Size of stack must be aligned to this (there are no notes about this in the api
120  // docs, but it will assert if you don't ...). Note this value is not the same as nn::os::FiberStackAlignment.
121  static const int stack_size_alignment = 0x1000;
122  stack_size = ((stack_size + stack_size_alignment - 1) / stack_size_alignment) * stack_size_alignment;
123 
124  m_fiber_context = m_memory_functions.user_alloc(stack_size, nn::os::FiberStackAlignment);
125  if (m_fiber_context == nullptr)
126  {
127  return result::out_of_memory;
128  }
129 
130  nn::os::InitializeFiber(&m_fiber_handle, trampoline_entry_point, reinterpret_cast<void*>(this), m_fiber_context, stack_size, 0);
131 
132  m_fiber_handle_created = true;
133 
134 #else
135 
136 # error Unimplemented platform
137 
138 #endif
139 
140  return result::success;
141 }
