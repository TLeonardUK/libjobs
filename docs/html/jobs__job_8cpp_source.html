51 result job_context::enter_scope(profile_scope_type type, bool unformatted, const char* tag, ...)
52 {
53  if (scheduler->m_profile_functions.leave_scope == nullptr)
54  {
55  return result::success;
56  }
57 
58  profile_scope_definition* scope = nullptr;
59  result res = scheduler->alloc_scope(scope);
60  if (res != result::success)
61  {
62  scheduler->write_log(debug_log_verbosity::error, debug_log_group::memory, "Failed to allocate profiling marker. Profiling results may be unprdictable.");
63  return res;
64  }
65 
66  if (unformatted)
67  {
68  strncpy(scope->tag, tag, profile_scope_definition::max_tag_length - 1);
69  scope->tag[profile_scope_definition::max_tag_length - 1] = '\0';
70  }
71  else
72  {
73  va_list list;
74  va_start(list, tag);
75  vsnprintf(scope->tag, profile_scope_definition::max_tag_length, tag, list);
76  va_end(list);
77  }
78 
79  scope->type = type;
80  scope->prev = profile_stack_tail;
81  scope->next = nullptr;
82 
83  if (profile_stack_head == nullptr)
84  {
85  profile_stack_head = scope;
86  }
87 
88  if (profile_stack_tail != nullptr)
89  {
90  profile_stack_tail->next = scope;
91  }
92  profile_stack_tail = scope;
93 
94  profile_scope_depth++;
95 
96  scheduler->m_profile_functions.enter_scope(scope->type, scope->tag);
97 
98  return result::success;
99 }

101 result job_context::leave_scope()
102 {
103  if (scheduler->m_profile_functions.leave_scope == nullptr)
104  {
105  return result::success;
106  }
107 
108  profile_scope_definition* original = profile_stack_tail;
109 
110  if (profile_stack_tail->prev != nullptr)
111  {
112  profile_stack_tail->prev->next = nullptr;
113  }
114  if (profile_stack_tail == profile_stack_head)
115  {
116  profile_stack_head = nullptr;
117  }
118 
119  profile_stack_tail = profile_stack_tail->prev;
120  profile_scope_depth--;
121 
122  scheduler->m_profile_functions.leave_scope();
123 
124  return scheduler->free_scope(original);
125 }

157 profile_scope_internal::profile_scope_internal(jobs::profile_scope_type type, const char* tag, jobs::scheduler* scheduler)
158 {
159  if (!scheduler::is_profiling_active())
160  {
161  return;
162  }
163 
164  internal::job_context* context = scheduler::get_active_job_context();
165  if (context != nullptr)
166  {
167  context->enter_scope(type, true, tag);
168  }
169  else if (scheduler != nullptr)
170  {
171  scheduler->m_profile_functions.enter_scope(type, tag);
172  }
173 
174  m_scheduler = scheduler;
175  m_context = context;
176 }

178 profile_scope_internal::~profile_scope_internal()
179 {
180  if (!scheduler::is_profiling_active())
181  {
182  return;
183  }
184 
185  internal::job_context* context = scheduler::get_active_job_context();
186 
187  // Not sure if this situation is even possible, but make sure we always
188  // leave on the same context we enter on.
189  assert(context == m_context);
190 
191  if (context != nullptr)
192  {
193  context->leave_scope();
194  }
195  else if (m_scheduler != nullptr)
196  {
197  m_scheduler->m_profile_functions.leave_scope();
